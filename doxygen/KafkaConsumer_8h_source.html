467 // Seek & Position
468 inline void
469 KafkaConsumer::seek(const TopicPartition& tp, Offset o, std::chrono::milliseconds timeout)
470 {
471  std::string tpStr = toString(tp);
472  KAFKA_API_DO_LOG(LOG_INFO, "will seek with topic-partition[%s], offset[%d]", tpStr.c_str(), o);
473 
474  auto rkt = rd_kafka_topic_unique_ptr(rd_kafka_topic_new(getClientHandle(), tp.first.c_str(), nullptr));
475  if (!rkt)
476  {
477  KAFKA_THROW_RESP_ERROR(rd_kafka_last_error());
478  }
479 
480  const auto end = std::chrono::steady_clock::now() + timeout;
481 
482  rd_kafka_resp_err_t err = RD_KAFKA_RESP_ERR_NO_ERROR;
483  do
484  {
485  err = rd_kafka_seek(rkt.get(), tp.second, o, SEEK_RETRY_INTERVAL_MS);
486  if (err != RD_KAFKA_RESP_ERR__STATE && err != RD_KAFKA_RESP_ERR__TIMED_OUT && err != RD_KAFKA_RESP_ERR__OUTDATED)
487  {
488  break;
489  }
490 
491  // If the "seek" was called just after "assign", there's a chance that the toppar's "fetch_state" (async setted) was not ready yes.
492  // If that's the case, we would retry again (normally, just after a very short while, the "seek" would succeed)
493  std::this_thread::yield();
494  } while (std::chrono::steady_clock::now() < end);
495 
496  KAFKA_THROW_IF_WITH_RESP_ERROR(err);
497 
498  KAFKA_API_DO_LOG(LOG_INFO, "seeked with topic-partition[%s], offset[%d]", tpStr.c_str(), o);
499 }

521 inline std::map<TopicPartition, Offset>
522 KafkaConsumer::offsetsForTime(const TopicPartitions& tps,
523  std::chrono::time_point<std::chrono::system_clock> timepoint,
524  std::chrono::milliseconds timeout) const
525 {
526  if (tps.empty()) return TopicPartitionOffsets();
527 
528  auto msSinceEpoch = std::chrono::duration_cast<std::chrono::milliseconds>(timepoint.time_since_epoch()).count();
529 
530  auto rk_tpos = rd_kafka_topic_partition_list_unique_ptr(createRkTopicPartitionList(tps));
531 
532  for (int i = 0; i < rk_tpos->cnt; ++i)
533  {
534  rd_kafka_topic_partition_t& rk_tp = rk_tpos->elems[i];
535  // Here the `msSinceEpoch` would be overridden by the offset result (after called by `rd_kafka_offsets_for_times`)
536  rk_tp.offset = msSinceEpoch;
537  }
538 
539  rd_kafka_resp_err_t err = rd_kafka_offsets_for_times(getClientHandle(), rk_tpos.get(), timeout.count());
540  KAFKA_THROW_IF_WITH_RESP_ERROR(err);
541 
542  auto results = getTopicPartitionOffsets(rk_tpos.get());
543 
544  // Remove invalid results (which are not updated with an valid offset)
545  for (auto it = results.begin(); it != results.end(); )
546  {
547  it = ((it->second == msSinceEpoch) ? results.erase(it) : std::next(it));
548  }
549 
550  return results;
551 }

626 // Fetch messages (internally used)
627 inline void
628 KafkaConsumer::pollMessages(int timeoutMs, std::vector<ConsumerRecord>& output)
629 {
630  // Commit the offsets for these messages which had been polled last time (for KafkaAutoCommitConsumer)
631  commitStoredOffsetsIfNecessary(CommitType::Async);
632 
633  // Poll messages with librdkafka's API
634  rd_kafka_message_t *msgPtrArray[_maxPollRecords];
635  std::size_t msgReceived = rd_kafka_consume_batch_queue(_rk_queue.get(), timeoutMs, msgPtrArray, _maxPollRecords);
636 
637  // Wrap messages with ConsumerRecord
638  output.clear();
639  output.reserve(msgReceived);
640  std::for_each(msgPtrArray, msgPtrArray + msgReceived, [&output](rd_kafka_message_t* rkMsg) { output.emplace_back(rkMsg); });
641 
642  // Store the offsets for all these polled messages (for KafkaAutoCommitConsumer)
643  storeOffsetsIfNecessary(output);
644 }

663 inline void
664 KafkaConsumer::pauseOrResumePartitions(const TopicPartitions& tps, PauseOrResumeOperation op)
665 {
666  auto rk_tpos = rd_kafka_topic_partition_list_unique_ptr(createRkTopicPartitionList(tps));
667 
668  rd_kafka_resp_err_t err = (op == PauseOrResumeOperation::Pause) ?
669  rd_kafka_pause_partitions(getClientHandle(), rk_tpos.get()) : rd_kafka_resume_partitions(getClientHandle(), rk_tpos.get());
670  KAFKA_THROW_IF_WITH_RESP_ERROR(err);
671 
672  const char* opString = (op == PauseOrResumeOperation::Pause) ? "pause" : "resume";
673  int cnt = 0;
674  for (int i = 0; i < rk_tpos->cnt; ++i)
675  {
676  const rd_kafka_topic_partition_t& rk_tp = rk_tpos->elems[i];
677  if (rk_tp.err != RD_KAFKA_RESP_ERR_NO_ERROR)
678  {
679  KAFKA_API_DO_LOG(LOG_ERR, "%s topic-partition[%s-%d] error[%s]", opString, rk_tp.topic, rk_tp.partition, rd_kafka_err2str(rk_tp.err));
680  }
681  else
682  {
683  KAFKA_API_DO_LOG(LOG_INFO, "%sd topic-partition[%s-%d]", opString, rk_tp.topic, rk_tp.partition, rd_kafka_err2str(rk_tp.err));
684  ++cnt;
685  }
686  }
687 
688  if (cnt == 0 && op == PauseOrResumeOperation::Pause)
689  {
690  std::string errMsg = std::string("No partition could be ") + opString + std::string("d among TopicPartitions[") + toString(tps) + std::string("]");
691  KAFKA_THROW_WITH_MSG(RD_KAFKA_RESP_ERR__INVALID_ARG, errMsg);
692  }
693 }

719 // Rebalance Callback (for class instance)
720 inline void
721 KafkaConsumer::onRebalance(rd_kafka_resp_err_t err, rd_kafka_topic_partition_list_t* rk_partitions)
722 {
723  TopicPartitions tps = getTopicPartitions(rk_partitions);
724  std::string tpsStr = toString(tps);
725 
726  switch(err)
727  {
728  case RD_KAFKA_RESP_ERR__ASSIGN_PARTITIONS:
729  KAFKA_API_DO_LOG(LOG_INFO, "invoked re-balance callback for event[ASSIGN_PARTITIONS]. topic-partitions[%s]", tpsStr.c_str());
730 
731  // Assign with a brand new full list
732  _assign(tps);
733  break;
734 
735  case RD_KAFKA_RESP_ERR__REVOKE_PARTITIONS:
736  KAFKA_API_DO_LOG(LOG_INFO, "invoked re-balance callback for event[REVOKE_PARTITIONS]. topic-partitions[%s]", tpsStr.c_str());
737 
738  _offsetsToStore.clear();
739 
740  // For "manual commit" cases, user must take all the responsibility to commit while necessary.
741  // -- thus, they must register a valid rebalance event listener and do the "commit things" properly.
742 
743  // Revoke all previously assigned partitions.
744  // Normally, another "ASSIGN_PARTITIONS" event would be received later.
745  _assign(TopicPartitions()); // with null
746  break;
747 
748  default:
749  KAFKA_API_DO_LOG(LOG_ERR, "invoked re-balance callback for event[unknown: %d]. topic-partitions[%s]", err, tpsStr.c_str());
750 
751  _assign(TopicPartitions()); // with null
752  return; // would not call user's rebalance event listener
753  }
754 
755  if (_rebalanceCb)
756  {
757  RebalanceEventType et =
758  (err == RD_KAFKA_RESP_ERR__ASSIGN_PARTITIONS ? RebalanceEventType::PartitionsAssigned : RebalanceEventType::PartitionsRevoked);
759  _rebalanceCb(et, tps);
760  }
761 }

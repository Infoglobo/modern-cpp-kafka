169 inline Properties
170 KafkaProducer::validateAndReformProperties(const Properties& origProperties)
171 {
172  // Let the base class validate first
173  Properties properties = KafkaClient::validateAndReformProperties(origProperties);
174 
175  // Check whether it's an available partitioner
176  const std::set<std::string> availPartitioners = {"murmur2_random", "murmur2", "random", "consistent", "consistent_random", "fnv1a", "fnv1a_random"};
177  auto partitioner = properties.getProperty(ProducerConfig::PARTITIONER);
178  if (partitioner && !availPartitioners.count(*partitioner))
179  {
180  std::string errMsg = "Invalid partitioner [" + *partitioner + "]! Valid options: ";
181  bool isTheFirst = true;
182  for (const auto& availPartitioner: availPartitioners)
183  {
184  errMsg += (std::string(isTheFirst ? (isTheFirst = false, "") : ", ") + availPartitioner);
185  }
186  errMsg += ".";
187 
188  KAFKA_THROW_ERROR(Error(RD_KAFKA_RESP_ERR__INVALID_ARG, errMsg));
189  }
190 
191  // For "idempotence" feature
192  constexpr int KAFKA_IDEMP_MAX_INFLIGHT = 5;
193  const auto enableIdempotence = properties.getProperty(ProducerConfig::ENABLE_IDEMPOTENCE);
194  if (enableIdempotence && *enableIdempotence == "true")
195  {
196  if (const auto maxInFlight = properties.getProperty(ProducerConfig::MAX_IN_FLIGHT))
197  {
198  if (std::stoi(*maxInFlight) > KAFKA_IDEMP_MAX_INFLIGHT)
199  {
200  KAFKA_THROW_ERROR(Error(RD_KAFKA_RESP_ERR__INVALID_ARG,\
201  "`max.in.flight` must be set <= " + std::to_string(KAFKA_IDEMP_MAX_INFLIGHT) + " when `enable.idempotence` is `true`"));
202  }
203  }
204 
205  if (const auto acks = properties.getProperty(ProducerConfig::ACKS))
206  {
207  if (*acks != "all" && *acks != "-1")
208  {
209  KAFKA_THROW_ERROR(Error(RD_KAFKA_RESP_ERR__INVALID_ARG,\
210  "`acks` must be set to `all`/`-1` when `enable.idempotence` is `true`"));
211  }
212  }
213  }
214 
215  return properties;
216 }

229 inline Error
230 KafkaProducer::sendMessage(const ProducerRecord& record,
231  std::unique_ptr<DeliveryCbOpaque> opaque,
232  SendOption option,
233  ActionWhileQueueIsFull action)
234 {
235  const auto* topic = record.topic().c_str();
236  const auto partition = record.partition();
237  const auto msgFlags = (static_cast<unsigned int>(option == SendOption::ToCopyRecordValue ? RD_KAFKA_MSG_F_COPY : 0)
238  | static_cast<unsigned int>(action == ActionWhileQueueIsFull::Block ? RD_KAFKA_MSG_F_BLOCK : 0));
239  const auto* keyPtr = record.key().data();
240  const auto keyLen = record.key().size();
241  const auto* valuePtr = record.value().data();
242  const auto valueLen = record.value().size();
243 
244  auto* rk = getClientHandle();
245  auto* opaquePtr = opaque.get();
246 
247  constexpr std::size_t VU_LIST_SIZE_WITH_NO_HEADERS = 6;
248  std::vector<rd_kafka_vu_t> rkVUs(VU_LIST_SIZE_WITH_NO_HEADERS + record.headers().size());
249 
250  std::size_t uvCount = 0;
251 
252  { // Topic
253  auto& vu = rkVUs[uvCount++];
254  vu.vtype = RD_KAFKA_VTYPE_TOPIC;
255  vu.u.cstr = topic;
256  }
257 
258  { // Partition
259  auto& vu = rkVUs[uvCount++];
260  vu.vtype = RD_KAFKA_VTYPE_PARTITION;
261  vu.u.i32 = partition;
262  }
263 
264  { // Message flags
265  auto& vu = rkVUs[uvCount++];
266  vu.vtype = RD_KAFKA_VTYPE_MSGFLAGS;
267  vu.u.i = static_cast<int>(msgFlags);
268  }
269 
270  { // Key
271  auto& vu = rkVUs[uvCount++];
272  vu.vtype = RD_KAFKA_VTYPE_KEY;
273  vu.u.mem.ptr = const_cast<void*>(keyPtr); // NOLINT
274  vu.u.mem.size = keyLen;
275  }
276 
277  { // Value
278  auto& vu = rkVUs[uvCount++];
279  vu.vtype = RD_KAFKA_VTYPE_VALUE;
280  vu.u.mem.ptr = const_cast<void*>(valuePtr); // NOLINT
281  vu.u.mem.size = valueLen;
282  }
283 
284  { // Opaque
285  auto& vu = rkVUs[uvCount++];
286  vu.vtype = RD_KAFKA_VTYPE_OPAQUE;
287  vu.u.ptr = opaquePtr;
288  }
289 
290  // Headers
291  for (const auto& header: record.headers())
292  {
293  auto& vu = rkVUs[uvCount++];
294  vu.vtype = RD_KAFKA_VTYPE_HEADER;
295  vu.u.header.name = header.key.c_str();
296  vu.u.header.val = header.value.data();
297  vu.u.header.size = header.value.size();
298  }
299 
300  assert(uvCount == rkVUs.size());
301 
302  Error error{ rd_kafka_produceva(rk, rkVUs.data(), rkVUs.size()) };
303 
304  // KafkaProducer::deliveryCallback would delete the "opaque"
305  if (!error) opaque.release();
306 
307  return error; // NOLINT: leak of memory pointed to by 'opaquePtr' [clang-analyzer-cplusplus.NewDeleteLeaks]
308 
309 }

517  {
518  Error deliveryResult;
519  std::promise<Producer::RecordMetadata> metadataPromise;
520  auto metadataFuture = metadataPromise.get_future();
521

523  deliveryResult = error;
524  metadataPromise.set_value(metadata);
525  };
526  Error sendResult{ sendMessage(record,
527  std::make_unique<DeliveryCbOpaque>(record.id(), cb),
528  SendOption::NoCopyRecordValue,
529  ActionWhileQueueIsFull::NoBlock) };
530  KAFKA_THROW_IF_WITH_ERROR(sendResult);
531 
532  while (metadataFuture.wait_for(std::chrono::milliseconds(0)) != std::future_status::ready)
533  {
534  rd_kafka_poll(getClientHandle(), CALLBACK_POLLING_INTERVAL_MS);
535  }
536 
537  auto metadata = metadataFuture.get();
538  KAFKA_THROW_IF_WITH_ERROR(deliveryResult);
539 
540  return metadata;

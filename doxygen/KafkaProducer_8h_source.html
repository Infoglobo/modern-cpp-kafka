201 inline Properties
202 KafkaProducer::validateAndReformProperties(const Properties& origProperties)
203 {
204  // Let the base class validate first
205  Properties properties = KafkaClient::validateAndReformProperties(origProperties);
206 
207  // Check whether it's an available partitioner
208  const std::set<std::string> availPartitioners = {"murmur2_random", "murmur2", "random", "consistent", "consistent_random", "fnv1a", "fnv1a_random"};
209  auto partitioner = properties.getProperty(ProducerConfig::PARTITIONER);
210  if (partitioner && !availPartitioners.count(*partitioner))
211  {
212  std::string errMsg = "Invalid partitioner [" + *partitioner + "]! Valid options: ";
213  bool isTheFirst = true;
214  for (const auto& availPartitioner: availPartitioners)
215  {
216  errMsg += (std::string(isTheFirst ? (isTheFirst = false, "") : ", ") + availPartitioner);
217  }
218  errMsg += ".";
219 
220  KAFKA_THROW_WITH_MSG(RD_KAFKA_RESP_ERR__INVALID_ARG, errMsg);
221  }
222 
223  // For "idempotence" feature
224  constexpr int KAFKA_IDEMP_MAX_INFLIGHT = 5;
225  const auto enableIdempotence = properties.getProperty(ProducerConfig::ENABLE_IDEMPOTENCE);
226  if (enableIdempotence && *enableIdempotence == "true")
227  {
228  if (const auto maxInFlight = properties.getProperty(ProducerConfig::MAX_IN_FLIGHT))
229  {
230  if (std::stoi(*maxInFlight) > KAFKA_IDEMP_MAX_INFLIGHT)
231  {
232  KAFKA_THROW_WITH_MSG(RD_KAFKA_RESP_ERR__INVALID_ARG,\
233  "`max.in.flight` must be set <= " + std::to_string(KAFKA_IDEMP_MAX_INFLIGHT) + " when `enable.idempotence` is `true`");
234  }
235  }
236 
237  if (const auto acks = properties.getProperty(ProducerConfig::ACKS))
238  {
239  if (*acks != "all" && *acks != "-1")
240  {
241  KAFKA_THROW_WITH_MSG(RD_KAFKA_RESP_ERR__INVALID_ARG,\
242  "`acks` must be set to `all`/`-1` when `enable.idempotence` is `true`");
243  }
244  }
245  }
246 
247  return properties;
248 }

261 inline rd_kafka_resp_err_t
262 KafkaProducer::sendMessage(const ProducerRecord& record,
263  std::unique_ptr<MsgOpaque> opaque,
264  SendOption option,
265  ActionWhileQueueIsFull action)
266 {
267  const auto* topic = record.topic().c_str();
268  const auto partition = record.partition();
269  const auto msgFlags = (static_cast<unsigned int>(option == SendOption::ToCopyRecordValue ? RD_KAFKA_MSG_F_COPY : 0)
270  | static_cast<unsigned int>(action == ActionWhileQueueIsFull::Block ? RD_KAFKA_MSG_F_BLOCK : 0));
271  const auto* keyPtr = record.key().data();
272  const auto keyLen = record.key().size();
273  const auto* valuePtr = record.value().data();
274  const auto valueLen = record.value().size();
275 
276  auto* rk = getClientHandle();
277  auto* opaquePtr = opaque.get();
278 
279  rd_kafka_resp_err_t sendResult = RD_KAFKA_RESP_ERR_NO_ERROR;
280 
281  if (auto cntHeaders = record.headers().size())
282  {
283  rd_kafka_headers_t* hdrs = rd_kafka_headers_new(cntHeaders);
284  for (const auto& header: record.headers())
285  {
286  rd_kafka_header_add(hdrs, header.key.c_str(), header.key.size(), header.value.data(), header.value.size());
287  }
288 
289 #pragma GCC diagnostic push
290 #pragma GCC diagnostic ignored "-Wold-style-cast"
291  sendResult = rd_kafka_producev(rk,
292  RD_KAFKA_V_TOPIC(topic),
293  RD_KAFKA_V_PARTITION(partition),
294  RD_KAFKA_V_MSGFLAGS(msgFlags),
295  RD_KAFKA_V_HEADERS(hdrs),
296  RD_KAFKA_V_VALUE(const_cast<void*>(valuePtr), valueLen), // NOLINT
297  RD_KAFKA_V_KEY(keyPtr, keyLen),
298  RD_KAFKA_V_OPAQUE(opaquePtr),
299  RD_KAFKA_V_END);
300 #pragma GCC diagnostic pop
301  if (sendResult != RD_KAFKA_RESP_ERR_NO_ERROR)
302  {
303  rd_kafka_headers_destroy(hdrs);
304  }
305  }
306  else
307  {
308 #pragma GCC diagnostic push
309 #pragma GCC diagnostic ignored "-Wold-style-cast"
310  sendResult = rd_kafka_producev(rk,
311  RD_KAFKA_V_TOPIC(topic),
312  RD_KAFKA_V_PARTITION(partition),
313  RD_KAFKA_V_MSGFLAGS(msgFlags),
314  RD_KAFKA_V_VALUE(const_cast<void*>(valuePtr), valueLen), // NOLINT
315  RD_KAFKA_V_KEY(keyPtr, keyLen),
316  RD_KAFKA_V_OPAQUE(opaquePtr),
317  RD_KAFKA_V_END);
318 #pragma GCC diagnostic pop
319  }
320 
321  if (sendResult == RD_KAFKA_RESP_ERR_NO_ERROR)
322  {
323  // KafkaProducer::deliveryCallback would delete the "opaque"
324  opaque.release();
325  }
326 
327  return sendResult; // NOLINT: leak of memory pointed to by 'opaquePtr' [clang-analyzer-cplusplus.NewDeleteLeaks]
328 }
